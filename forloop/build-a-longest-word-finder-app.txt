function findLongestWordLength(str){
const split = str.split(' ')
const longestStr = split.reduce((max, word) => word.length > max ? word.length : max, 0)
return longestStr
}

const output = findLongestWordLength("The quick brown fox jumped over the lazy dog")
console.log(output)
