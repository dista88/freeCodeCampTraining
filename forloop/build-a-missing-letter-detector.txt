function fearNotLetter(letter) {
 for(let i = 1; i < letter.length; i++){
   let firstIndex = letter[i].charCodeAt(0)
   let lastIndex =  letter[i-1].charCodeAt(0)
   if(firstIndex - lastIndex > 1){
    return String.fromCharCode(lastIndex + 1)
   }
 }
 
}
