function largestOfAll(arr){
  const arrayMax = []
for(let i = 0; i < arr.length; i++){
  const maxInArray = Math.max(...arr[i])
  arrayMax.push(maxInArray)
}
return arrayMax
}

const output = largestOfAll([[4, 5, 1, 3], [13, 27, 18, 26], [32, 35, 37, 39], [1000, 1001, 857, 1]])

console.log(output)
