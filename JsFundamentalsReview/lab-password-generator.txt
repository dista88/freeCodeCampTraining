function generatePassword(lengthPass){
  const passwordChar = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()"
  let password = '';

  // const passwordLength = lengthPass > 0 ? lengthPass : 12;

  for(let i = 0; i < lengthPass; i++){
    const randomIndex = Math.floor(Math.random() * passwordChar.length)
    password += passwordChar[randomIndex]
  }
  return password
 
}

const password = generatePassword(1)

console.log(`Generated password: ${password}`)
