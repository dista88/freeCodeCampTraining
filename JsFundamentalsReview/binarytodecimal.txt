function toDecimal(binary) {
  let split = binary.split("")

  let decimal = 0
  let power = 0
  for(let i = split.length - 1; i>=0; i--){
  let digit = Number(split[i])
  decimal += digit * Math.pow(2, power) 
  power++
   } 
  return decimal
}
